# python entreda.py <endereço_entrada> <porta_entrada>

# python <endereço_entrada> <porta_entrada> <endereço_host1> <porta_host1>
# python <endereço_entrada> <porta_entrada> <endereço_host2> <porta_host2>
...
# python <endereço_entrada> <porta_entrada> <endereço_hostN> <porta_hostN>
